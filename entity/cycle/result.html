Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\CycleManager;

$manager = new CycleManager();

$result = $manager->create([
    "started_at" => "2019-06-27 10:30:06",
    "started_status" => "activation",
    "ended_At" => "2019-06-27 10:30:06",
    "ended_status" => "termination",
    "cyclable_type" => "foo",
    "cyclable" => [
        "name" => "Lorenzo Wunsch",
        "description" => "Voluptatibus molestias numquam aut quas veritatis molestiae. Et error autem consectetur reiciendis sed consequatur hic.",
        "bar" => [
            "name" => "Prof. Lenore O'Keefe Sr.",
            "description" => "Quia soluta delectus et harum. Ipsam ratione minima explicabo iusto velit distinctio. Laudantium maiores consequatur ab repellat. Iure quasi ut veritatis sit facere at similique."
        ]
    ]
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
